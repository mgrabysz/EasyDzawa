klasa Ułamek {
    
  Ułamek(l, m) {            // konstruktor
    tenże.licznik = l;
    tenże.mianownik = m;
    jeżeli (m == 0) {
        kończWaść();
    }
    jeżeli (l < m) {
        tenże.jestWłaściwy = prawda;    // ułamek właściwy to taki, który jest mniejszy od 1
    } inaczej {
        tenże.jestWłaściwy = fałsz;
    }
  }
}
